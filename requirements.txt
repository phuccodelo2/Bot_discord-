-- Tọa độ gốc (nút cửa)

-- Tọa độ trên cao (cách mặt đất 10m)
local abovePos = groundPos + Vector3.new(0, 2, 0)

-- Tele lên trên cao
hrp.CFrame = CFrame.new(abovePos)

print("✅ Đã tele lên trên cao. Đang rơi xuống...")
