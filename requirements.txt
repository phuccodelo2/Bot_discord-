-- 🧍‍♂️ Fake bóng nhân vật đứng tại vị trí chỉ định
local Players = game:GetService("Players")
local lp = Players.LocalPlayer
local char = lp.Character or lp.CharacterAdded:Wait()

-- Toạ độ muốn tạo bóng
local targetPos = Vector3.new(-517.5, -5.1, 101.4)

-- Tạo bản sao nhân vật
local clone = char:Clone()
clone.Name = "FakeShadow"
clone.Parent = workspace

-- Dừng mọi hoạt động
if clone:FindFirstChild("Humanoid") then
	clone.Humanoid:ChangeState(Enum.HumanoidStateType.Physics)
end

-- Gỡ mọi script để tránh bug
for _, v in pairs(clone:GetDescendants()) do
	if v:IsA("Script") or v:IsA("LocalScript") then
		v:Destroy()
	end
end

-- Đặt vị trí đứng yên
if clone:FindFirstChild("HumanoidRootPart") then
	clone:MoveTo(targetPos + Vector3.new(0, 3, 0)) -- Đứng cao hơn mặt đất một chút
end

-- Làm cho trong suốt 30% để giống bóng
for _, part in pairs(clone:GetDescendants()) do
	if part:IsA("BasePart") then
		part.Anchored = true
		part.CanCollide = false
		part.Transparency = 0.3
	end
end

print("✅ Đã tạo bóng đứng tại tọa độ.")
