-- ⚙️ TELE FAKE (CHẠM NÚT TỪ XA)
-- Toạ độ cần chạm:
local targetPos = Vector3.new(-517.5, -5.1, 101.4)

-- Tìm nhân vật và HumanoidRootPart
local lp = game:GetService("Players").LocalPlayer
local char = lp.Character or lp.CharacterAdded:Wait()
local hrp = char:WaitForChild("HumanoidRootPart")

-- Tìm part gần tọa độ đó để giả chạm
local nearestPart = nil
local shortestDist = math.huge

for _, part in ipairs(workspace:GetDescendants()) do
    if part:IsA("BasePart") and part.CanTouch and (part.Position - targetPos).Magnitude < 10 then
        local dist = (part.Position - targetPos).Magnitude
        if dist < shortestDist then
            shortestDist = dist
            nearestPart = part
        end
    end
end

-- Gửi sự kiện chạm (touch) nếu tìm được part gần đó
if nearestPart then
    firetouchinterest(hrp, nearestPart, 0)
    task.wait(0.1)
    firetouchinterest(hrp, nearestPart, 1)
    print("✅ Đã tele fake đến vị trí và chạm part:", nearestPart.Name)
else
    warn("❌ Không tìm thấy part gần tọa độ để chạm.")
end
