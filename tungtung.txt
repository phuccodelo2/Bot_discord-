-- üõ°Ô∏è B·∫•t t·ª≠ auto khi b·∫≠t script
local function makeGod()
    local hum = char:FindFirstChildOfClass("Humanoid")
    if hum then
        hum.MaxHealth = math.huge
        hum.Health = math.huge
        hum:GetPropertyChangedSignal("Health"):Connect(function()
            if hum.Health < math.huge then
                hum.Health = math.huge
            end
        end)
    end
end
makeGod()

end)

-- üè† To·∫° ƒë·ªô c·ª≠a
local doorPositions = {
    Vector3.new(-469, -7, -102),
    Vector3.new(-468, -7, 8),
    Vector3.new(-467, -7, 112),
    Vector3.new(-466, -8, 220),
    Vector3.new(-355, -8, 219),
    Vector3.new(-354, -8, 112),
    Vector3.new(-353, -7, 4),
    Vector3.new(-353, -7, -100)
}
local function getNearestDoor()
    local closest, minDist = nil, math.huge
    for _, door in ipairs(doorPositions) do
        local dist = (root.Position - door).Magnitude
        if dist < minDist then
            minDist = dist
            closest = door
        end
    end
    return closest
end

-- ‚òÅÔ∏è Bay l√™n 200m
local function goUp()
    local door = getNearestDoor()
    if door then
        TweenService:Create(root, TweenInfo.new(1.2), {CFrame = CFrame.new(door)}):Play()
        wait(1.3)
        root.CFrame = root.CFrame + Vector3.new(0, 200, 0)
        StarterGui:SetCore("ChatMakeSystemMessage", {
            Text = "[PhucMax ‚òÅÔ∏è] Bay l√™n 200m!",
            Color = Color3.fromRGB(0,255,128),
            TextSize = 18
        })
    end
end

-- ‚õì R∆°i xu·ªëng 50m
local function dropDown()
    root.CFrame = root.CFrame - Vector3.new(0, 50, 0)
    StarterGui:SetCore("ChatMakeSystemMessage", {
        Text = "[PhucMax ‚õì] R∆°i xu·ªëng th√†nh c√¥ng!",

local function toggleInvisible(on)
    for _, v in pairs(char:GetDescendants()) do
        if v:IsA("BasePart") then v.Transparency = on and 1 or 0 end
    end
    pcall(function()
        char.Head:FindFirstChild("Nametag").Enabled = not on
    end)
end

-- üíÄ Toggle b·∫•t t·ª≠ l·∫°i

            wait(0.5)
            char:BreakJoints()
        end)
    end
end

local function createToggle(text, callback)
    local btn = Instance.new("TextButton", content)
    btn.Size = UDim2.new(1, -20, 0, 30)
    btn.Position = UDim2.new(0, 10, 0, #content:GetChildren() * 32 - 32)
    btn.BackgroundColor3 = Color3.fromRGB(100, 0, 0)
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = Enum.Font.GothamBold
    btn.TextSize = 14
    btn.Text = "‚ùå " .. text
    btn.AutoButtonColor = false
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 6)
    local on = false
    btn.MouseButton1Click:Connect(function()
        on = not on
        btn.BackgroundColor3 = on and Color3.fromRGB(0, 170, 0) or Color3.fromRGB(170, 0, 0)
        btn.Text = (on and "‚úÖ " or "‚ùå ") .. text
        callback(on)
    end)
end

-- üåà Vi·ªÅn v√† ch·ªØ ƒë·ªïi m√†u
local function rainbowify(obj)
    local h = 0
    RunService.RenderStepped:Connect(function()
        h = (h + 0.01) % 1
        local color = Color3.fromHSV(h, 1, 1)
        if obj:IsA("TextLabel") then
            obj.TextColor3 = color
        else
            obj.BackgroundColor3 = color
        end
    end)
end
